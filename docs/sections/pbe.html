viewof filtro_setor = Inputs.select(["Todos", ...new Set(demandas_proto.map(d => d.Setor).filter(Boolean))], {label: "Setor:"})
viewof filtro_status = Inputs.select(["Todos", ...new Set(demandas_proto.map(d => d.Status).filter(Boolean))], {label: "Status:"})

tarefas_filtradas = demandas_proto.filter(d => 
  (filtro_setor === "Todos" || d.Setor === filtro_setor) &&
  (filtro_status === "Todos" || d.Status === filtro_status)
)

// Lógica de cálculo de dias
function calcularDiasRestantes(prazoStr) {
  if (!prazoStr || prazoStr.trim() === "") return null;
  const partes = prazoStr.split(/[-/]/);
  let dataPrazo = partes.length === 3 ? new Date(partes[0], partes[1] - 1, partes[2]) : new Date(prazoStr);
  const hoje = new Date();
  hoje.setHours(0,0,0,0);
  dataPrazo.setHours(0,0,0,0);
  return Math.ceil((dataPrazo - hoje) / (1000 * 60 * 60 * 24));
}
// Renderização HTML personalizada para o tema Cyborg
html`<div style="display: flex; flex-direction: column; gap: 12px; margin-top: 15px;">
  ${tarefas_filtradas.map(d => {
    const dias = d.Prazo ? calcularDiasRestantes(d.Prazo) : null;
    let prazoTexto = "Sem prazo";
    let badgeCor = "#444"; 
    
    if (dias !== null) {
      if (dias < 0) { prazoTexto = `Atrasada (${Math.abs(dias)}d)`; badgeCor = "#dc3545"; }
      else if (dias === 0) { prazoTexto = "Entrega HOJE"; badgeCor = "#fd7e14"; }
      else { prazoTexto = `${dias} dias restantes`; badgeCor = dias <= 5 ? "#ffc107" : "#198754"; }
      
      // Se concluído, sobrepõe a cor do prazo
      if (d.Status === "Concluído") badgeCor = "#444";
    }

    return `
      <details style="background: #1e1e1e; border-left: 4px solid ${d.Status === 'Concluído' ? '#198754' : d.Status === 'Em andamento' ? '#0dcaf0' : '#dc3545'}; border-radius: 6px; padding: 12px 16px; box-shadow: 0 4px 6px rgba(0,0,0,0.3);">
        <summary style="cursor: pointer; font-weight: 500; font-size: 1.05em; display: flex; justify-content: space-between; align-items: center; list-style: none;">
          <div style="display: flex; align-items: center; gap: 10px;">
            <span style="color: #adb5bd; font-size: 0.85em; text-transform: uppercase; letter-spacing: 0.5px;">${d.Setor}</span>
            <span style="color: #fff;">${d.Tarefa}</span>
          </div>
          <div style="display: flex; gap: 8px;">
            <span style="background: ${badgeCor}; color: ${badgeCor === '#ffc107' ? '#000' : '#fff'}; padding: 4px 10px; border-radius: 20px; font-size: 0.75em; font-weight: bold;">
              ${d.Status === 'Concluído' ? 'Finalizado' : prazoTexto}
            </span>
