📋 Quadro de Tarefas

viewof filtro_setor = Inputs.select(["Todos", ...new Set(demandas_proto.map(d => d.Setor).filter(Boolean))], {label: "Setor:"})
viewof filtro_status = Inputs.select(["Todos", ...new Set(demandas_proto.map(d => d.Status).filter(Boolean))], {label: "Status:"})
function calcularDiasRestantes(prazoStr) {
  if (!prazoStr || !prazoStr.trim()) return null;
  const partes = prazoStr.split(/[-/]/);
  let dataPrazo = partes.length === 3 ? new Date(partes[0], partes[1] - 1, partes[2]) : new Date(prazoStr);
  const hoje = new Date(); hoje.setHours(0,0,0,0); dataPrazo.setHours(0,0,0,0);
  return Math.ceil((dataPrazo - hoje) / (86400000));
}

// Filtro + Ordenação por urgência (com ou sem prazo)
tarefas_filtradas = demandas_proto
  .filter(d => 
    (filtro_setor === "Todos" || d.Setor === filtro_setor) &&
    (filtro_status === "Todos" || d.Status === filtro_status)
  )
  .sort((a, b) => {
    let diasA = calcularDiasRestantes(a.Prazo);
    let diasB = calcularDiasRestantes(b.Prazo);
    let pesoStatus = { "Pendente": 1, "Faltando só o detalhe": 2, "Concluído": 3 };
    
    if (diasA !== null && diasB !== null) return diasA - diasB;
    if (diasA !== null) return -1;
    if (diasB !== null) return 1;
    
    return (pesoStatus[a.Status] || 4) - (pesoStatus[b.Status] || 4);
  })
html`<div>
  ${tarefas_filtradas.map(d => {
    const dias = calcularDiasRestantes(d.Prazo);
    let prazoTexto = "Sem prazo", badgeCor = "#444"; 
    
    if (dias !== null) {
      if (dias < 0) { prazoTexto = `Atrasada (${Math.abs(dias)}d)`; badgeCor = "#dc3545"; }
      else if (dias === 0) { prazoTexto = "Entrega HOJE"; badgeCor = "#fd7e14"; }
      else { prazoTexto = `${dias} dias restantes`; badgeCor = dias <= 5 ? "#ffc107" : "#198754"; }
      if (d.Status === "Concluído") badgeCor = "#444";
    }

    let corBorda = d.Status === 'Concluído' ? '#198754' : d.Status === 'Faltando só o detalhe' ? '#ffc107' : '#dc3545';
    let txtBorda = d.Status === 'Faltando só o detalhe' ? '#000' : '#fff';

    // Condicional para o Detalhe
    let detalheHtml = (d.Status === 'Faltando só o detalhe' && d.Detalhe)
