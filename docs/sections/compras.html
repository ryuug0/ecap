dados_compras = d3.csv("https://docs.google.com/spreadsheets/d/e/2PACX-1vR_kABBziuLQDcgkwtThORXGuhxlp9G1mWXsb9Xv6XzB9T4K47YFWr6oRXahriBjqsIeQgl1pe_stIe/pub?gid=210558369&single=true&output=csv")

// Tratamento dos valores monetários
compras_tratadas = dados_compras.map(d => {
  let valorLimpo = 0;
  if (d.Valor) {
    valorLimpo = parseFloat(d.Valor.replace("R$", "").replace(/\./g, "").replace(",", ".").trim()) || 0;
  } else if (d.Preço && d.Quantidade) {
    let p = parseFloat(d.Preço.replace("R$", "").replace(/\./g, "").replace(",", ".").trim()) || 0;
    let q = parseInt(d.Quantidade) || 1;
    valorLimpo = p * q;
  }
  return { ...d, valor_num: valorLimpo };
})

// Divisão proporcional para o gráfico
compras_expandidas = compras_tratadas.flatMap(d => {
  if (!d.Categoria) return [];
  let categorias = d.Categoria.split(",").map(c => c.trim());
  let numCategorias = categorias.length;
  return categorias.map(c => ({
    ...d,
    Categoria: c,
    valor_proporcional: d.valor_num / numCategorias
  }));
})

viewof filtro_proj = Inputs.select(["Todos", "Protótipo", "Conceito Urbano", "Geral"], {label: "Filtrar Projeto:", value: "Todos"})
viewof filtro_stat = Inputs.select(["Todos", "Aguardando verba", "Em cotação", "Concluído"], {label: "Filtrar Status:", value: "Todos"})
itens_filtrados = compras_tratadas
  .filter(d => 
    (filtro_proj === "Todos" || (d.Categoria && d.Categoria.split(",").map(c=>c.trim()).includes(filtro_proj))) &&
    (filtro_stat === "Todos" || d.Status === filtro_stat)
  )
  .sort((a, b) => {
    let pesoPrioridade = { "Alta": 1, "Média": 2, "Baixa": 3 };
    return (pesoPrioridade[a.Prioridade] || 4) - (pesoPrioridade[b.Prioridade] || 4);
  })
