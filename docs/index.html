// 2. Lógica de Datas e Prazos (Corrigida)
function parseData(prazoStr) {
  if (!prazoStr || !prazoStr.trim()) return null;
  let p = prazoStr.trim().split(/[-/]/);
  let dia, mes, ano;
  if (p.length === 2) { 
    dia = parseInt(p[0]); mes = parseInt(p[1]) - 1; ano = new Date().getFullYear();
  } else if (p.length === 3) {
    if (p[0].length === 4) { ano = parseInt(p[0]); mes = parseInt(p[1]) - 1; dia = parseInt(p[2]); }
    else { dia = parseInt(p[0]); mes = parseInt(p[1]) - 1; ano = parseInt(p[2]); }
    if (ano < 100) ano += 2000;
  } else { return null; }
  let d = new Date(ano, mes, dia);
  return isNaN(d.getTime()) ? null : d;
}

function calcularDiasRestantes(prazoStr) {
  let dataPrazo = parseData(prazoStr);
  if (!dataPrazo) return null;
  const hoje = new Date(); hoje.setHours(0,0,0,0); dataPrazo.setHours(0,0,0,0);
  return Math.ceil((dataPrazo - hoje) / 86400000);
}

function formatarDataBR(prazoStr) {
  let d = parseData(prazoStr);
  return d ? d.toLocaleDateString('pt-BR', {day: '2-digit', month: '2-digit', year: 'numeric'}) : "---";
}

viewof filtro_global_status = Inputs.select(["Todos", "Pendente", "Faltando só o detalhe", "Concluído"], {label: "Status:", value: "Todos"})
tarefas_filtradas = dados_gerais
  .filter(d => 
    (filtro_global_projeto === "Todos" || d.Categoria === filtro_global_projeto) &&
    (filtro_global_status === "Todos" || d.Status === filtro_global_status)
  )
  .sort((a, b) => {
    let diasA = calcularDiasRestantes(a.Prazo);
    let diasB = calcularDiasRestantes(b.Prazo);
    let pesoStatus = { "Pendente": 1, "Faltando só o detalhe": 2, "Concluído": 3 };
    if (diasA !== null && diasB !== null) return diasA - diasB;
    if (diasA !== null) return -1;
    if (diasB !== null) return 1;
    return (pesoStatus[a.Status] || 4) - (pesoStatus[b.Status] || 4);
  })
